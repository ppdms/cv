% Define a clean environment for bullet lists with precise spacing
% Adjust 'itemsep' to control space between bullets. 1.5pt is tight but breathing.
% Hanging indent: the dash sits in the left gutter as a scan anchor and
% wrapped lines align with the text — that shape is what keeps bullets
% distinguishable. Text is justified (full lines align to the page's right
% rule, same as the dates) but never hyphenated; microtype's font expansion
% absorbs the spacing adjustment. If a line ever turns gappy, reword it.
\newenvironment{resumeitemize}{
  \begin{itemize}[label={\color{applegray}--}, leftmargin=0.15in, itemsep=2.5pt, topsep=0pt, parsep=0pt, partopsep=0pt]
  \hyphenpenalty=10000\exhyphenpenalty=10000
  \tolerance=2000 \emergencystretch=1em
}{
  \end{itemize}\vspace{-3pt}
}

% Line ends must be %-protected: an unprotected newline leaves a trailing
% space in the bullet paragraph, and when the last line is flush-full TeX is
% forced to break at that space — yielding a phantom zero-height extra line.
\newcommand{\resumeItem}[1]{%
 \item\small{%
  {#1}%
 }%
}
